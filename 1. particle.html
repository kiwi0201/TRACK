        def inform(unit, lot, pr, c_rcp, i_rcp, s_list, pre_list, post_list):
            slot_count = len(s_list)
            pre_count = len(pre_list)
            post_count = len(post_list)
            indv_slot = ""
        

            if not (1 <= slot_count <= 25):

                return "slot 개수는 1~25의 정수여야 합니다."

        

            if (slot_count != pre_count or slot_count != post_count or pre_count != post_count):

                return "#slot, #pre, #post가 매칭되지 않습니다. 다시 확인해주세요."

        

            for i in range(0,slot_count):

                temp_1 = s_list[i]

                indv_slot += str(temp_1)+','

        

            indv_slot = indv_slot[:-1]

        

            info = f"{lot} #{indv_slot} (PR : {pr} / {c_rcp} / {i_rcp})\n"

        

            for i in range(slot_count):

                slot_num = s_list[i]

                pre = pre_list[i]

                post = post_list[i]

                if i % 4 == 0:

                    info += f"{unit} 11-01 #{slot_num} : {post-pre}ea ( {pre}ea → {post}ea )\n"

                elif i % 4 == 1:

                    info += f"{unit} 11-02 #{slot_num} : {post-pre}ea ( {pre}ea → {post}ea )\n"

                elif i % 4 == 2:

                    info += f"{unit} 12-01 #{slot_num} : {post-pre}ea ( {pre}ea → {post}ea )\n"

                elif i % 4 == 3:

                    info += f"{unit} 12-02 #{slot_num} : {post-pre}ea ( {pre}ea → {post}ea )\n\n"

        

            return info

    

        unit = input("module : ")

        lot = input("lot : ")

        pr = input("PR : ")

        c_rcp = input("COATING RCP : ")

        i_rcp = input("INSPECTION RCP : ")

        s_list = list(map(int, input("slot list : ").split(',')))

        pre_list = list(map(int, input("pre : ").split(',')))

        post_list = list(map(int, input("post : ").split(',')))

        

        result = inform(unit, lot, pr, c_rcp, i_rcp, s_list, pre_list, post_list)

        print(result)
